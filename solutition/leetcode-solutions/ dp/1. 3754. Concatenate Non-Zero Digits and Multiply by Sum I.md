```
  class Solution {
    public long sumAndMultiply(int n) {
        String s = String.valueOf(n);
        long sum = 0;
        StringBuilder sb = new StringBuilder();

        for (char c : s.toCharArray()) {
            int digit = c - '0';
            sum += digit;
            if (digit != 0) {
                sb.append(c);
            }
        }

        long x = (sb.length() == 0) ? 0 : Long.parseLong(sb.toString());

        return x * sum;
    }
}

```
```
 class Solution {
    public long sumAndMultiply(int n) {
        String s = String.valueOf(n);
        long sum = 0;
        StringBuilder sb = new StringBuilder();

        for (char c : s.toCharArray()) {
            int digit = Character.getNumericValue(c); // NO c - '0'

            sum += digit;

            if (digit != 0) {
                sb.append(c); // keep original character
            }
        }

        long x = (sb.length() == 0) ? 0 : Long.parseLong(sb.toString());

        return x * sum;
    }
}
```
