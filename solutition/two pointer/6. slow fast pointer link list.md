static class Node {

// Floyd's Cycle Detection
public static boolean hasCycle(Node head) {
    Node slow = head, fast = head;
    while (fast != null && fast.next != null) {
        slow = slow.next;
        fast = fast.next.next;
        if (slow == fast) return true;
    }
    return false;
}

public static Node detectCycleStart(Node head) {
    Node slow = head, fast = head;

    // Step 1: detect intersection
    while (fast != null && fast.next != null) {
        slow = slow.next;
        fast = fast.next.next;
        if (slow == fast) {
            // Step 2: move slow to head
            slow = head;
            while (slow != fast) {
                slow = slow.next;
                fast = fast.next;
            }
            return slow;
        }
    }
    return null;
}

public static int cycleLength(Node meetingPoint) {
    Node curr = meetingPoint;
    int len = 0;
    do {
        curr = curr.next;
        len++;
    } while (curr != meetingPoint);
    return len;
}
